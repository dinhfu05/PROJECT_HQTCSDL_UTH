CÁC BẢN DỮ LIỆU MẪU: 

USERS:
- userID (PK)
- username: varchar(50)
- email: varchar(100)
- password_hash: varchar(255)
- created_at: datetime
- updated_at: datetime

PRODUCTS
- productId (PK)
- productName: varchar(100)
- productDescription: text
- productPrice: decimal
- Category_id (FK)
- created_at: datetime
- updated_at: datetime

ORDERS
- id (PK)
- user_id (FK)
- status: varchar(50) => enums ('pending', 'shipped', 'delivered', 'canceled')
- total_amount: decimal
- created_at: datetime
- updated_at: datetime

ORDER_ITEMS
- id (PK)
- order_id (FK)
- product_id (FK)
- quantity: integer
- price: decimal

CATEGORIES
- id (PK)
- name: varchar(100)
